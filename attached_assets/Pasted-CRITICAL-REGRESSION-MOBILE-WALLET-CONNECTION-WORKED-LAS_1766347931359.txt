import { useEffect } from 'react';

function App() {
  // Prevent background scroll when wallet modal is open
  useEffect(() => {
    const observer = new MutationObserver(() => {
      const modal = document.querySelector('div[data-rk][role="dialog"]');
      
      if (modal) {
        // Modal is open - prevent scroll
        document.body.style.overflow = 'hidden';
        document.body.style.position = 'fixed';
        document.body.style.width = '100%';
        document.body.style.top = '0';
        console.log('ğŸ”’ Modal open - scroll locked');
      } else {
        // Modal is closed - restore scroll
        document.body.style.overflow = '';
        document.body.style.position = '';
        document.body.style.width = '';
        document.body.style.top = '';
        console.log('ğŸ”“ Modal closed - scroll unlocked');
      }
    });
    
    observer.observe(document.body, { 
      childList: true, 
      subtree: true 
    });
    
    return () => {
      observer.disconnect();
      // Cleanup on unmount
      document.body.style.overflow = '';
      document.body.style.position = '';
      document.body.style.width = '';
      document.body.style.top = '';
    };
  }, []);
  
  // ... rest of App component
}

        >
          {address ? `${address.slice(0, 6)}...${address.slice(-4)}` : 'Connect Wallet'}
        </button>
        
        <CustomWalletModal 
          isOpen={showModal} 
          onClose={() => setShowModal(false)} 
        />
      </>
    );
  }
  
  // Desktop: use RainbowKit
  return <ConnectButton />;
}
