import { useState, useCallback } from 'react';
import { useAccount, useWriteContract, useWaitForTransactionReceipt } from 'wagmi';
import { parseEther } from 'viem';
import { NFT_CONTRACT, RPC_URL } from '@/lib/constants';

const MINT_PRICE = 69420; // $BASED per NFT

const NFT_ABI = [
  {
    name: 'mint',
    type: 'function',
    stateMutability: 'payable',
    inputs: [{ name: 'quantity', type: 'uint256' }],
    outputs: [],
  },
  {
    name: 'totalMinted',
    type: 'function',
    stateMutability: 'view',
    inputs: [],
    outputs: [{ type: 'uint256' }],
  },
] as const;

export function useMint() {
  const { address, isConnected } = useAccount();
  const [isLoading, setIsLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const [txHash, setTxHash] = useState<string | null>(null);

  const { writeContract, data: hash, isPending, reset } = useWriteContract();

  const { isLoading: isConfirming, isSuccess } = useWaitForTransactionReceipt({
    hash: hash,
  });

  const mint = useCallback(async (quantity: number = 1) => {
    if (!isConnected || !address) {
      setError('Please connect your wallet');
      return;
    }

    try {
      setIsLoading(true);
      setError(null);
      setTxHash(null);

      const totalPrice = MINT_PRICE * quantity;
      
      console.log('ðŸš€ Minting NFT...');
      console.log('   Quantity:', quantity);
      console.log('   Total Price:', totalPrice, '$BASED');
      console.log('   Contract:', NFT_CONTRACT);

      writeContract({
        address: NFT_CONTRACT as `0x${string}`,
        abi: NFT_ABI,
        functionName: 'mint',
        args: [BigInt(quantity)],
        value: parseEther(totalPrice.toString()),
        gas: BigInt(500000), // Explicit gas limit - THIS WAS KEY
      });

    } catch (e: any) {
      console.error('Mint error:', e);
      setError(e?.message || 'Failed to mint');
      setIsLoading(false);
    }
  }, [isConnected, address, writeContract]);

  // Update loading state based on transaction status
  const loading = isLoading || isPending || isConfirming;

  // Reset function
  const resetMint = useCallback(() => {
    setIsLoading(false);
    setError(null);
    setTxHash(null);
    reset();
  }, [reset]);

  return {
    mint,
    isLoading: loading,
    isPending,
    isConfirming,
    isSuccess,
    error,
    txHash: hash,
    reset: resetMint,
    mintPrice: MINT_PRICE,
  };
}

export default useMint;

KEY POINTS THAT MADE IT WORK:
1. gas: BigInt(500000) - Explicit gas limit, don't let it estimate
2. parseEther(totalPrice.toString()) - Proper value formatting
3. Simple ABI with just 'mint' function
4. No fancy gas price overrides - let the network handle it

DO NOT ADD:
- gasPrice overrides (was causing issues)
- maxFeePerGas / maxPriorityFeePerGas (EIP-1559 was broken)
- Complex retry logic
- Extra validation that might block the transaction

This is the SIMPLE version that worked for minting #149.
