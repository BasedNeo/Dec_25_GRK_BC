  // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
  // DECK INITIALIZATION
  // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•

  const initializeDeck = useCallback((): SolitaireCard[] => {
    const newDeck: SolitaireCard[] = [];
    const usedNFTs = new Set<number>();
    
    for (let suit = 0; suit < 4; suit++) {
      for (let rank = 1; rank <= 13; rank++) {
        let nftId: number;
        do {
          nftId = Math.floor(Math.random() * 3732) + 1;
        } while (usedNFTs.has(nftId) && usedNFTs.size < 3732);
        
        usedNFTs.add(nftId);
        
        newDeck.push({
          id: `${SUITS[suit]}-${rank}-${nftId}`,
          nftId,
          suit: SUITS[suit],
          rank,
          faceUp: false,
          imageUrl: `${IPFS_ROOT}${nftId}.png`
        });
      }
    }
    
    // Fisher-Yates shuffle
    for (let i = newDeck.length - 1; i > 0; i--) {
      const j = Math.floor(Math.random() * (i + 1));
      [newDeck[i], newDeck[j]] = [newDeck[j], newDeck[i]];
    }
    
    return newDeck;
  }, []);

  // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
  // DEAL CARDS
  // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•

  const dealCards = useCallback(() => {
    const shuffledDeck = initializeDeck();
    const newTableau: Pile[] = Array(7).fill(null).map(() => ({ cards: [] }));
    let deckIndex = 0;

    for (let i = 0; i < 7; i++) {
      for (let j = 0; j <= i; j++) {
        const card = { ...shuffledDeck[deckIndex++] };
        if (j === i) card.faceUp = true;
        newTableau[i].cards.push(card);
      }
    }

    const remainingDeck = shuffledDeck.slice(deckIndex).map(c => ({ ...c, faceUp: false }));

    setTableau(newTableau);
    setDeck(remainingDeck);
    setWaste([]);
    setFoundations(Array(4).fill(null).map(() => ({ cards: [] })));
    setMoves(0);
    setInvalidAttempts(0);
    setStartTime(Date.now());
    setElapsedTime(0);
    setGameWon(false);
    setSelectedCard(null);
    setGameHistory([]);
    setComboCount(0);
    playSound('flip');
  }, [initializeDeck, playSound]);

  // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
  // STATE HISTORY (UNDO)
  // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•

  const saveStateToHistory = useCallback(() => {
    const state: GameState = {
      deck: JSON.parse(JSON.stringify(deck)),
      waste: JSON.parse(JSON.stringify(waste)),
      foundations: JSON.parse(JSON.stringify(foundations)),
      tableau: JSON.parse(JSON.stringify(tableau)),
      moves,
      startTime: startTime || Date.now(),
      invalidAttempts,
    };
    
    setGameHistory(prev => {
      const newHistory = [...prev, state];
      return newHistory.slice(-MAX_UNDO_HISTORY);
    });
  }, [deck, waste, foundations, tableau, moves, startTime, invalidAttempts]);

  // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
  // UNDO MOVE
  // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•

  const undoMove = useCallback(() => {
    if (gameHistory.length === 0) {
      toast({ 
        title: "No Moves to Undo", 
        description: "You're at the start of the game",
        variant: "destructive" 
      });
      return;
    }
    
    const previousState = gameHistory[gameHistory.length - 1];
    setDeck(previousState.deck);
    setWaste(previousState.waste);
    setFoundations(previousState.foundations);
    setTableau(previousState.tableau);
    setMoves(previousState.moves);
    setInvalidAttempts(previousState.invalidAttempts);
    setGameHistory(prev => prev.slice(0, -1));
    playSound('slide');
    
    trackEvent('game_action', 'Solitaire', 'Undo Move');
  }, [gameHistory, playSound, toast]);

  // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
  // VALIDATION
  // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•

  const canPlaceOnTableau = useCallback((card: SolitaireCard, targetPile: Pile): boolean => {
    if (targetPile.cards.length === 0) {
      return card.rank === 13; // Only Kings
    }
    const topCard = targetPile.cards[targetPile.cards.length - 1];
    const differentColor = 
      (['hearts', 'diamonds'].includes(card.suit) !== ['hearts', 'diamonds'].includes(topCard.suit));
    return differentColor && card.rank === topCard.rank - 1;
  }, []);

  const canPlaceOnFoundation = useCallback((card: SolitaireCard, foundation: Pile): boolean => {
    if (foundation.cards.length === 0) {
      return card.rank === 1; // Ace starts
    }
    const topCard = foundation.cards[foundation.cards.length - 1];
    return card.suit === topCard.suit && card.rank === topCard.rank + 1;
  }, []);

  // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
  // FIND VALID MOVES (HINT)
  // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•

  const findValidMoves = useCallback(() => {
    const moves: { from: string; to: string; card: SolitaireCard; priority: number }[] = [];
    
    // Priority: Foundation > Expose cards > General moves
    
    if (waste.length > 0) {
      foundations.forEach((f, idx) => {
        if (canPlaceOnFoundation(waste[0], f)) {
          moves.push({ from: 'waste', to: `foundation-${idx}`, card: waste[0], priority: 3 });
        }
      });
    }
    
    tableau.forEach((pile, pileIdx) => {
      if (pile.cards.length > 0) {
        const topCard = pile.cards[pile.cards.length - 1];
        if (topCard.faceUp) {
          foundations.forEach((f, fIdx) => {
            if (canPlaceOnFoundation(topCard, f)) {
              moves.push({ from: `tableau-${pileIdx}`, to: `foundation-${fIdx}`, card: topCard, priority: 3 });
            }
          });
        }
      }
    });
    
    tableau.forEach((pile, pileIdx) => {
      if (pile.cards.length > 1) {
        const topCard = pile.cards[pile.cards.length - 1];
        const secondCard = pile.cards[pile.cards.length - 2];
        if (topCard.faceUp && !secondCard.faceUp) {
          tableau.forEach((targetPile, targetIdx) => {
            if (targetIdx !== pileIdx && canPlaceOnTableau(topCard, targetPile)) {
              moves.push({ from: `tableau-${pileIdx}`, to: `tableau-${targetIdx}`, card: topCard, priority: 2 });
            }
          });
        }
      }
    });
    
    if (waste.length > 0) {
      tableau.forEach((pile, idx) => {
        if (canPlaceOnTableau(waste[0], pile)) {
          moves.push({ from: 'waste', to: `tableau-${idx}`, card: waste[0], priority: 1 });
        }
      });
    }
    
    moves.sort((a, b) => b.priority - a.priority);
    return moves;
  }, [waste, foundations, tableau, canPlaceOnFoundation, canPlaceOnTableau]);

  // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
  // SHOW HINT
  // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•

  const showHintMove = useCallback(() => {
    const validMoves = findValidMoves();
    
    if (validMoves.length === 0) {
      toast({ 
        title: "No Immediate Moves", 
        description: deck.length > 0 ? "Try drawing from the deck" : "No moves available",
        variant: "destructive" 
      });
      playSound('invalid');
      return;
    }
    
    setShowHint(true);
    playSound('combo');
    
    const bestMove = validMoves[0];
    const priority = bestMove.priority === 3 ? "Foundation move available" : 
                    bestMove.priority === 2 ? "Move to expose card" : "Valid move found";
    
    toast({
      title: "Hint",
      description: `${priority} (${validMoves.length} move${validMoves.length > 1 ? 's' : ''} total)`,
      className: "bg-black border-yellow-500 text-yellow-400"
    });
    
    setTimeout(() => setShowHint(false), 3000);
    trackEvent('game_action', 'Solitaire', 'Used Hint');
  }, [findValidMoves, deck.length, playSound, toast]);

  // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
  // DRAW FROM DECK
  // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•

  const drawCard = useCallback(() => {
    saveStateToHistory();
    
    if (deck.length === 0) {
      if (waste.length === 0) {
        toast({
          title: "Deck Empty",
          description: "No cards left to draw",
          variant: "destructive"
        });
        return;
      }
      setDeck(waste.reverse().map(c => ({ ...c, faceUp: false })));
      setWaste([]);
      playSound('flip');
    } else {
      const newCard = { ...deck[0], faceUp: true };
      setWaste([newCard, ...waste]);
      setDeck(deck.slice(1));
      setMoves(m => m + 1);
      playSound('flip');
    }
    
    trackEvent('game_action', 'Solitaire', 'Draw Card');
  }, [deck, waste, saveStateToHistory, playSound, toast]);

  // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
  // MOVE TO TABLEAU
  // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•

  const moveToTableau = useCallback((targetIndex: number) => {
    if (!selectedCard) return;
    
    const targetPile = tableau[targetIndex];
    
    if (!canPlaceOnTableau(selectedCard.card, targetPile)) {
      setInvalidAttempts(prev => prev + 1);
      playSound('invalid');
      
      let reason = "Invalid move";
      if (targetPile.cards.length === 0) {
        reason = "Only Kings can go on empty piles";
      } else {
        const topCard = targetPile.cards[targetPile.cards.length - 1];
        const sameColor = (['hearts', 'diamonds'].includes(selectedCard.card.suit) === ['hearts', 'diamonds'].includes(topCard.suit));
        if (sameColor) {
          reason = "Must alternate red and black cards";
        } else {
          reason = "Must place in descending order";
        }
      }
      
      toast({ 
        title: "Invalid Move", 
        description: reason,
        variant: "destructive" 
      });
      setSelectedCard(null);
      return;
    }

    saveStateToHistory();
    createParticles(window.innerWidth / 2, window.innerHeight / 2, 12);

    const newTableau = [...tableau];

    if (selectedCard.from === 'waste') {
      newTableau[targetIndex] = { cards: [...targetPile.cards, selectedCard.card] };
      setWaste(waste.slice(1));
    } else if (selectedCard.from === 'tableau') {
      const sourcePile = tableau[selectedCard.index];
      const cardIndex = sourcePile.cards.findIndex(c => c.id === selectedCard.card.id);
      const cardsToMove = sourcePile.cards.slice(cardIndex);
      
      newTableau[selectedCard.index] = { cards: sourcePile.cards.slice(0, cardIndex) };
      
      if (newTableau[selectedCard.index].cards.length > 0) {
        const lastCard = newTableau[selectedCard.index].cards[newTableau[selectedCard.index].cards.length - 1];
        if (!lastCard.faceUp) {
          lastCard.faceUp = true;
          playSound('flip');
        }
      }
      
      newTableau[targetIndex] = { cards: [...targetPile.cards, ...cardsToMove] };
    }

    setTableau(newTableau);
    setMoves(m => m + 1);
    setSelectedCard(null);
    playSound('snap');
    
    setComboCount(0);
    if (comboTimerRef.current) clearTimeout(comboTimerRef.current);
    
    trackEvent('game_action', 'Solitaire', 'Move to Tableau');
  }, [selectedCard, tableau, waste, canPlaceOnTableau, playSound, toast, saveStateToHistory, createParticles]);

  // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
  // MOVE TO FOUNDATION
  // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•

  const moveToFoundation = useCallback((foundationIndex: number) => {
    if (!selectedCard) return;
    
    const targetFoundation = foundations[foundationIndex];
    
    if (!canPlaceOnFoundation(selectedCard.card, targetFoundation)) {
      setInvalidAttempts(prev => prev + 1);
      playSound('invalid');
      
      const reason = targetFoundation.cards.length === 0 
        ? "Foundation must start with an Ace" 
        : "Must follow suit in ascending order";
      
      toast({ 
        title: "Invalid Move", 
        description: reason,
        variant: "destructive" 
      });
      setSelectedCard(null);
      return;
    }

    saveStateToHistory();
    createParticles(window.innerWidth / 2, 150, 18);

    const newFoundations = [...foundations];
    newFoundations[foundationIndex] = { 
      cards: [...targetFoundation.cards, selectedCard.card] 
    };
    setFoundations(newFoundations);

    if (selectedCard.from === 'waste') {
      setWaste(waste.slice(1));
    } else if (selectedCard.from === 'tableau') {
      const sourcePile = tableau[selectedCard.index];
      const newTableau = [...tableau];
      newTableau[selectedCard.index] = { cards: sourcePile.cards.slice(0, -1) };
      
      if (newTableau[selectedCard.index].cards.length > 0) {
        const lastCard = newTableau[selectedCard.index].cards[newTableau[selectedCard.index].cards.length - 1];
        if (!lastCard.faceUp) {
          lastCard.faceUp = true;
          playSound('flip');
        }
      }
      
      setTableau(newTableau);
    }

    setMoves(m => m + 1);
    
    // Combo system
    const newComboCount = comboCount + 1;
    setComboCount(newComboCount);
    
    if (comboTimerRef.current) clearTimeout(comboTimerRef.current);
    comboTimerRef.current = setTimeout(() => setComboCount(0), COMBO_TIMEOUT);
    
    setSelectedCard(null);
    playSound('combo');
    
    if (newComboCount >= 5) {
      toast({
        title: `${newComboCount}x Combo!`,
        description: "Foundation combo streak!",
        className: "bg-black border-orange-500 text-orange-400"
      });
    }
    
    trackEvent('game_action', 'Solitaire', 'Move to Foundation');
    
    // Check win
    setTimeout(() => {
      const allComplete = newFoundations.every(f => f.cards.length === 13);
      if (allComplete && !gameWon) {
        setGameWon(true);
        setShowVictory(true);
        endGame(true);
      }
    }, 100);
  }, [selectedCard, foundations, waste, tableau, comboCount, gameWon, canPlaceOnFoundation, playSound, toast, saveStateToHistory, createParticles]);

  // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
  // DOUBLE-CLICK AUTO-MOVE
  // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•

  const handleDoubleClick = useCallback((card: SolitaireCard, from: 'waste' | 'tableau', index: number) => {
    for (let i = 0; i < foundations.length; i++) {
      if (canPlaceOnFoundation(card, foundations[i])) {
        setSelectedCard({ card, from, index });
        setTimeout(() => moveToFoundation(i), 0);
        return;
      }
    }
    
    toast({
      title: "No Auto-Move Available",
      description: "Card cannot be placed on any foundation",
      variant: "destructive"
    });
  }, [foundations, canPlaceOnFoundation, moveToFoundation, toast]);

  // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
  // START GAME
  // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•

  const startGame = useCallback((resumeSaved: boolean = false) => {
    if (!isConnected) {
      toast({ 
        title: "Wallet Required", 
        description: "Please connect your wallet to play", 
        variant: "destructive" 
      });
      return;
    }
    
    if (!ownsNFT) {
      toast({ 
        title: "Guardian NFT Required", 
        description: "You must own a Guardian NFT to access this game", 
        variant: "destructive" 
      });
      return;
    }
    
    if (playsToday >= gameConfig.maxPlaysPerDay) {
      toast({ 
        title: "Daily Limit Reached", 
        description: `You've used all ${gameConfig.maxPlaysPerDay} plays today. Return tomorrow!`, 
        variant: "destructive" 
      });
      return;
    }

    if (resumeSaved && address) {
      const savedGame = GameStorageManager.loadSave<GameState>('guardian-solitaire', address);
      if (savedGame) {
        const state = savedGame.state;
        setDeck(state.deck);
        setWaste(state.waste);
        setFoundations(state.foundations);
        setTableau(state.tableau);
        setMoves(state.moves);
        setInvalidAttempts(state.invalidAttempts);
        setStartTime(state.startTime);
        setGameStarted(true);
        playSound('flip');
        
        toast({
          title: "Game Resumed",
          description: "Your previous game has been restored",
          className: "bg-black border-cyan-500 text-cyan-400"
        });
        return;
      }
    }

    setGameStarted(true);
    dealCards();
    trackEvent('game_start', 'Game', 'Guardian Solitaire');
  }, [isConnected, ownsNFT, playsToday, gameConfig.maxPlaysPerDay, address, dealCards, playSound, toast]);

  // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
  // END GAME

    const baseScore = won ? 10000 : 0;
    const timeBonus = won ? Math.max(0, 500 - finalTime) : 0;
    const moveBonus = won ? Math.max(0, 1000 - (moves * 5)) : 0;
    const comboBonus = comboCount * 100;
    const efficiency = moves > 0 ? Math.round((moves / (moves + invalidAttempts)) * 100) : 100;
    const efficiencyBonus = won ? Math.round(efficiency * 2) : 0;
    
    const finalScore = Math.min(
      baseScore + timeBonus + moveBonus + comboBonus + efficiencyBonus,
      gameConfig.scoring.maxScore
    );

    // Update stats
    const newStats: GameStats = { ...stats };
    newStats.gamesPlayed++;
    newStats.totalMoves += moves;
    newStats.totalTime += finalTime;
    
    if (won) {
      newStats.gamesWon++;
      newStats.currentStreak++;
      newStats.longestStreak = Math.max(newStats.longestStreak, newStats.currentStreak);
      
      if (finalTime < newStats.bestTime || newStats.bestTime === 0) {
        newStats.bestTime = finalTime;
      }
      if (moves < (newStats.bestMoves || Infinity)) {
        newStats.bestMoves = moves;
      }
      if (finalScore > newStats.bestScore) {
        newStats.bestScore = finalScore;
      }
      
      playSound('victory');
      createParticles(window.innerWidth / 2, window.innerHeight / 2, 60);
    } else {
      newStats.currentStreak = 0;
    }
    
    newStats.lastPlayed = Date.now();
    newStats.totalScore += finalScore;
    
    GameStorageManager.saveStats('guardian-solitaire', address, newStats);
    setStats(newStats);

    try {
      await submitScore(address, 'guardian-solitaire', finalScore);
      setPlaysToday(prev => prev + 1);
      trackEvent('game_complete', 'Game', `Solitaire - ${won ? 'Won' : 'Lost'} - ${finalScore} pts`);

        title: "Score Save Failed",
        description: "Could not save your score. Check your connection.",
        variant: "destructive"
      });

                    if (cardToMove) {
                      setSelectedCard(cardToMove);
                      setTimeout(() => moveToTableau(pileIdx), 0);
                    }
                  }
                }}
                onDrop={() => {
                  if (draggedCard) {
                    setSelectedCard(draggedCard);
                    setTimeout(() => moveToTableau(pileIdx), 0);
                  }
                }}
              />
            ))}
          </div>
        </div>
      </div>

      {/* Victory Screen */}
      {showVictory && (
        <VictoryScreen
          gameType="guardian-solitaire"
          score={(() => {
            const finalTime = Math.floor((Date.now() - (startTime || Date.now())) / 1000);
            const baseScore = 10000;
            const timeBonus = Math.max(0, 500 - finalTime);
            const moveBonus = Math.max(0, 1000 - (moves * 5));
            const comboBonus = comboCount * 100;
            const efficiency = moves > 0 ? Math.round((moves / (moves + invalidAttempts)) * 100) : 100;
            const efficiencyBonus = Math.round(efficiency * 2);
            return Math.min(
              baseScore + timeBonus + moveBonus + comboBonus + efficiencyBonus,
              gameConfig.scoring.maxScore
            );
          })()}
          time={elapsedTime}
          moves={moves}
          playsRemaining={gameConfig.maxPlaysPerDay - playsToday}
          maxPlays={gameConfig.maxPlaysPerDay}
          isNewBest={(() => {
            const finalTime = Math.floor((Date.now() - (startTime || Date.now())) / 1000);
            const score = 10000 + Math.max(0, 500 - finalTime) + Math.max(0, 1000 - (moves * 5));
            return score > stats.bestScore;
          })()}
          personalBest={stats.bestScore}
          onPlayAgain={() => {
            setShowVictory(false);
            setGameStarted(false);
          }}
          onExit={() => navigate('/')}
          formatTime={formatTime}
        />
      )}
    </section>
  );
}
