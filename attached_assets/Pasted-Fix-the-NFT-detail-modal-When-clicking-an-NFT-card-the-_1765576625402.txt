JAVASCRIPT:
function openNftModal(tokenId, name, image, owner, attributes) {
  const modal = document.getElementById('nftModal');
  
  // Set image
  document.getElementById('modalImage').src = image;
  
  // Set title
  document.getElementById('modalTitle').textContent = name || 'Based Guardian #' + tokenId;
  document.getElementById('modalSubtitle').textContent = 'Token ID: ' + tokenId;
  
  // Set owner
  const ownerEl = document.getElementById('modalOwner');
  const shortOwner = owner.substring(0, 6) + '...' + owner.substring(owner.length - 4);
  ownerEl.textContent = shortOwner;
  ownerEl.href = 'https://explorer.bf1337.org/address/' + owner;
  
  // Set attributes
  const attrContainer = document.getElementById('modalAttributes');
  attrContainer.innerHTML = '';
  
  if (attributes && attributes.length > 0) {
    attributes.forEach(attr => {
      const div = document.createElement('div');
      div.className = 'attribute-item';
      div.innerHTML = `
        <div class="attribute-type">${attr.trait_type}</div>
        <div class="attribute-value">${attr.value}</div>
      `;
      attrContainer.appendChild(div);
    });
  } else {
    attrContainer.innerHTML = '<p style="color:#666;">No attributes available</p>';
  }
  
  // Show modal
  modal.classList.add('active');
  document.body.style.overflow = 'hidden';
}

function closeNftModal() {
  const modal = document.getElementById('nftModal');
  modal.classList.remove('active');
  document.body.style.overflow = '';
}

function closeModal(event) {
  if (event.target.classList.contains('modal')) {
    closeNftModal();
  }
}

// Close on Escape key
document.addEventListener('keydown', (e) => {
  if (e.key === 'Escape') closeNftModal();
});
