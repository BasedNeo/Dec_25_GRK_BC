  const totalMinted = await contract.totalMinted();
  const nfts = [];
  const distribution = {
    "Rarest-Legendary": 0,
    "Very Rare": 0,
    "More Rare": 0,        // â† Note: "More Rare" not "Rarest"
    "Rare": 0,
    "Less Rare": 0,
    "Less Common": 0,
    "Common": 0,
    "Most Common": 0
  };
  
  // Fetch in parallel
  const promises = [];
  for (let i = 0; i < totalMinted; i++) {
    promises.push(fetchNFTData(i));
  }
  const results = await Promise.all(promises);
  
  results.forEach(nft => {
    nfts.push(nft);
    distribution[nft.rarity]++;
  });
  
  return { nfts, distribution, totalMinted };
}
