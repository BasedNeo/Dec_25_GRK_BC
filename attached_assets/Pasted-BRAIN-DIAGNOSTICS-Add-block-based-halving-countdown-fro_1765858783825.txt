ADD function to get BasedAI L1 block info:

async function getBasedAIBlockInfo() {
  try {
    const provider = new ethers.JsonRpcProvider(BRAIN_CONFIG.basedaiRpc);
    
    // Get current block number
    const currentBlock = await provider.getBlockNumber();
    
    // Get latest block for timestamp
    const latestBlock = await provider.getBlock('latest');
    const currentTimestamp = latestBlock?.timestamp || Math.floor(Date.now() / 1000);
    
    // Calculate blocks since genesis
    const genesisBlock = 0; // Block 0
    const blocksSinceGenesis = currentBlock - genesisBlock;
    
    // Calculate blocks until next halving
    const blocksInCurrentEpoch = blocksSinceGenesis % BRAIN_CONFIG.halvingInterval;
    const blocksUntilHalving = BRAIN_CONFIG.halvingInterval - blocksInCurrentEpoch;
    
    // Convert to time
    const secondsUntilHalving = blocksUntilHalving * BRAIN_CONFIG.blockTime;
    const daysUntilHalving = Math.floor(secondsUntilHalving / 86400);
    const hoursRemaining = Math.floor((secondsUntilHalving % 86400) / 3600);
    
    // Calculate halving epoch (0 = pre-halving, 1 = first halving, etc.)
    const currentEpoch = Math.floor(blocksSinceGenesis / BRAIN_CONFIG.halvingInterval);
    
    // Estimate halving date
    const halvingTimestamp = new Date(Date.now() + (secondsUntilHalving * 1000));
    
    return {
      currentBlock,
      blocksUntilHalving,
      daysUntilHalving,
      hoursRemaining,
      currentEpoch,
      halvingDate: halvingTimestamp.toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' }),
      networkTimestamp: currentTimestamp
    };
  } catch (e) {
    console.error('Failed to fetch BasedAI block info:', e);
    // Fallback to estimated values
    return {
      currentBlock: null,
      blocksUntilHalving: null,
      daysUntilHalving: 15, // Fallback estimate
      hoursRemaining: 0,
      currentEpoch: 0,
      halvingDate: '~Dec 31, 2025',
      networkTimestamp: null
    };
  }
}
