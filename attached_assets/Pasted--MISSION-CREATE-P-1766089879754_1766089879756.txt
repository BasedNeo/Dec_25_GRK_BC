const getDistance = (a: Vector2D, b: Vector2D): number => {
  const dx = a.x - b.x;
  const dy = a.y - b.y;
  return Math.sqrt(dx * dx + dy * dy);
};

const wrapPosition = (pos: Vector2D, margin: number = 0): Vector2D => {
  return {
    x: pos.x < -margin ? CANVAS_WIDTH + margin : pos.x > CANVAS_WIDTH + margin ? -margin : pos.x,
    y: pos.y < -margin ? CANVAS_HEIGHT + margin : pos.y > CANVAS_HEIGHT + margin ? -margin : pos.y,
  };
};

const randomInRange = (min: number, max: number): number => {
  return Math.random() * (max - min) + min;
};

const randomColor = (): AsteroidColor => {
  const rand = Math.random();
  if (rand < GOLD_ASTEROID_CHANCE) return 'gold';
  if (rand < 0.12) return 'purple';
  if (rand < 0.32) return 'green';
  if (rand < 0.62) return 'blue';
  return 'grey';
};

    });
  }, []);

  const splitAsteroid = useCallback((asteroid: Asteroid) => {
    const state = gameStateRef.current;
    
    if (asteroid.size === 'large') {
      // Split into 2 medium
      for (let i = 0; i < 2; i++) {
        const angle = (Math.PI * 2 * i) / 2 + Math.random() * 0.5;
        const speed = Math.sqrt(asteroid.velocity.x ** 2 + asteroid.velocity.y ** 2) * 1.2;
        
        state.asteroids.push({
          id: `asteroid-${Date.now()}-${i}`,
          position: { ...asteroid.position },
          velocity: {
            x: Math.cos(angle) * speed,
            y: Math.sin(angle) * speed,
          },
          size: 'medium',
          color: asteroid.color,
          rotation: 0,
          rotationSpeed: randomInRange(-3, 3),
          radius: ASTEROID_SIZES.medium.radius,
          health: ASTEROID_SIZES.medium.health,
        });
      }
    } else if (asteroid.size === 'medium') {
      // Split into 4 small
      for (let i = 0; i < 4; i++) {
        const angle = (Math.PI * 2 * i) / 4 + Math.random() * 0.3;
        const speed = Math.sqrt(asteroid.velocity.x ** 2 + asteroid.velocity.y ** 2) * 1.3;
        
        state.asteroids.push({
          id: `asteroid-${Date.now()}-${i}`,
          position: { ...asteroid.position },
          velocity: {
            x: Math.cos(angle) * speed,
            y: Math.sin(angle) * speed,
          },
          size: 'small',
          color: asteroid.color,
          rotation: 0,
          rotationSpeed: randomInRange(-4, 4),
          radius: ASTEROID_SIZES.small.radius,
          health: ASTEROID_SIZES.small.health,
        });
      }
    }
  }, []);

  const destroyAsteroid = useCallback((asteroid: Asteroid, hitByBullet: boolean = false) => {
    const state = gameStateRef.current;
    
    // Calculate points
    const basePoints = ASTEROID_SIZES[asteroid.size].points;
    const colorMultiplier = ASTEROID_COLOR_MULTIPLIERS[asteroid.color];
    let points = Math.floor(basePoints * colorMultiplier);
    
    // Apply combo multiplier
    if (state.combo >= 2) {
      points *= Math.min(state.combo, 4); // Max 4x
    }
    
    state.score += points;
    state.asteroidsDestroyed++;
    
    if (hitByBullet) {
      state.accuracy.hits++;
    }
    
    // Update combo
    state.combo++;
    state.comboTimer = COMBO_TIMEOUT;
    
    // Particles
    createParticles(asteroid.position, ASTEROID_COLORS[asteroid.color], 15, 150);
    
    // Drop resource
    const resourceValue = Math.floor(points * 0.5);
    state.resources.push({
      id: `resource-${Date.now()}`,
      position: { ...asteroid.position },
      velocity: { x: 0, y: 50 },
      color: asteroid.color,
      value: resourceValue,
      lifetime: 10,
    });
    
    // Drop power-up chance
    if (Math.random() < POWER_UP_DROP_CHANCE) {
      const powerUpTypes: PowerUpType[] = ['shield', 'rapidFire', 'magnet'];
      const type = powerUpTypes[Math.floor(Math.random() * powerUpTypes.length)];
      
      state.powerUps.push({
        id: `powerup-${Date.now()}`,
        position: { ...asteroid.position },
        velocity: { x: 0, y: 30 },
        type,
        lifetime: 15,
      });
    }
    
    // Extra life drop (rare)
    if (Math.random() < EXTRA_LIFE_DROP_CHANCE) {
      state.powerUps.push({
        id: `powerup-life-${Date.now()}`,
        position: { ...asteroid.position },
        velocity: { x: 0, y: 30 },
        type: 'extraLife',
        lifetime: 15,
      });
    }
    
    // Split or remove
    if (asteroid.size !== 'small' && asteroid.color !== 'gold') {
      splitAsteroid(asteroid);
    }
    
    // Remove asteroid
    state.asteroids = state.asteroids.filter(a => a.id !== asteroid.id);
    
    playSound('explosion');
  }, [createParticles, splitAsteroid, playSound]);

  const shoot = useCallback(() => {
    const state = gameStateRef.current;
    
    if (state.bullets.length >= MAX_BULLETS) return;
    
    const ship = state.ship;
    const fireRate = state.powerUpActive?.type === 'rapidFire' ? 3 : 1;
    
    for (let i = 0; i < fireRate; i++) {
      const offsetAngle = fireRate > 1 ? (i - 1) * 0.2 : 0;
      const angle = ship.rotation + offsetAngle;
      
      state.bullets.push({
        id: `bullet-${Date.now()}-${i}`,
        position: {
          x: ship.position.x + Math.cos(angle) * SHIP_SIZE,
          y: ship.position.y + Math.sin(angle) * SHIP_SIZE,
        },
        velocity: {
          x: Math.cos(angle) * BULLET_SPEED,
          y: Math.sin(angle) * BULLET_SPEED,
        },
        lifetime: BULLET_LIFETIME,
      });
    }
    
    state.accuracy.shots += fireRate;
    playSound('shoot');
  }, [playSound]);

  const damageShip = useCallback(() => {
    const state = gameStateRef.current;
    const ship = state.ship;
    
    if (ship.invulnerable) return;
    
    ship.shield--;
    ship.invulnerable = true;
    ship.invulnerableTime = INVULNERABLE_TIME;
    
    playSound('hit');
    createParticles(ship.position, '#EF4444', 10, 100);
    
    if (ship.shield <= 0) {
      ship.lives--;
      
      if (ship.lives > 0) {
        ship.shield = ship.maxShield;
        ship.position = { x: CANVAS_WIDTH / 2, y: CANVAS_HEIGHT / 2 };
        ship.velocity = { x: 0, y: 0 };
        toast({
          title: "Life Lost",
          description: `${ship.lives} ${ship.lives === 1 ? 'life' : 'lives'} remaining`,
          className: "bg-black border-red-500 text-red-400"
        });
      } else {
        setGameOver(true);
        setShowVictory(true);
        playSound('death');
        endGame();
      }
    }
  }, [playSound, createParticles, toast]);

  // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
  // GAME LOOP
  // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•

  const update = useCallback((deltaTime: number) => {
    if (isPaused || gameOver) return;
    
    const state = gameStateRef.current;
    const dt = deltaTime / 1000; // Convert to seconds
    
    // Update survival time
    state.survivalTime += deltaTime;
    
    // Update ship
    const ship = state.ship;
    
    // Move toward target position (mobile tap control)
    if (targetPosition) {
      const dx = targetPosition.x - ship.position.x;
      const dy = targetPosition.y - ship.position.y;
      const distance = Math.sqrt(dx * dx + dy * dy);
      
      if (distance > 5) {
        ship.rotation = Math.atan2(dy, dx);
        const speed = SHIP_SPEED * dt;
        ship.velocity.x = (dx / distance) * Math.min(speed, distance);
        ship.velocity.y = (dy / distance) * Math.min(speed, distance);
      } else {
        ship.velocity.x = 0;
        ship.velocity.y = 0;
      }
    }
    
    ship.position.x += ship.velocity.x;
    ship.position.y += ship.velocity.y;
    ship.position = wrapPosition(ship.position, SHIP_SIZE);
    
    // Update invulnerability
    if (ship.invulnerable) {
      ship.invulnerableTime -= deltaTime;
      if (ship.invulnerableTime <= 0) {
        ship.invulnerable = false;
      }
    }
    
    // Auto-shoot if button held
    if (isShooting) {
      const fireInterval = state.powerUpActive?.type === 'rapidFire' ? 100 : 250;
      if (state.survivalTime % fireInterval < 50) {
        shoot();
      }
    }
    
    // Update bullets
    state.bullets = state.bullets.filter(bullet => {
      bullet.position.x += bullet.velocity.x * dt;
      bullet.position.y += bullet.velocity.y * dt;
      bullet.lifetime -= dt;
      
      // Remove if off-screen or expired
      if (bullet.lifetime <= 0) return false;
      if (bullet.position.x < -50 || bullet.position.x > CANVAS_WIDTH + 50) return false;
      if (bullet.position.y < -50 || bullet.position.y > CANVAS_HEIGHT + 50) return false;
      
      return true;
    });
    
    // Update asteroids
    state.asteroids.forEach(asteroid => {
      asteroid.position.x += asteroid.velocity.x * dt;
      asteroid.position.y += asteroid.velocity.y * dt;
      asteroid.position = wrapPosition(asteroid.position, asteroid.radius + 50);
      asteroid.rotation += asteroid.rotationSpeed * dt;
    });
    
    // Update resources
    state.resources = state.resources.filter(resource => {
      resource.position.y += resource.velocity.y * dt;
      resource.lifetime -= dt;
      
      // Collect if magnet active
      if (state.powerUpActive?.type === 'magnet') {
        const dist = getDistance(resource.position, ship.position);
        if (dist < 150) {
          const dx = ship.position.x - resource.position.x;
          const dy = ship.position.y - resource.position.y;
          resource.velocity.x = (dx / dist) * 200;
          resource.velocity.y = (dy / dist) * 200;
        }
      }
      
      // Collect on contact
      const dist = getDistance(resource.position, ship.position);
      if (dist < SHIP_SIZE + 10) {
        state.score += resource.value;
        state.resourcesCollected++;
        playSound('collect');
        createParticles(resource.position, ASTEROID_COLORS[resource.color], 8, 80);
        return false;
      }
      
      return resource.lifetime > 0;
    });
    
    // Update power-ups
    state.powerUps = state.powerUps.filter(powerUp => {
      powerUp.position.y += powerUp.velocity.y * dt;
      powerUp.lifetime -= dt;
      
      // Collect on contact
      const dist = getDistance(powerUp.position, ship.position);
      if (dist < SHIP_SIZE + 15) {
        if (powerUp.type === 'extraLife') {
          ship.lives++;
          toast({
            title: "Extra Life!",
            description: `Now at ${ship.lives} lives`,
            className: "bg-black border-green-500 text-green-400"
          });
        } else if (powerUp.type === 'shield') {
          ship.shield = ship.maxShield;
          toast({
            title: "Shield Restored!",
            className: "bg-black border-cyan-500 text-cyan-400"
          });
        } else {
          state.powerUpActive = { type: powerUp.type, duration: POWER_UP_DURATION };
          const names = { rapidFire: 'Rapid Fire', magnet: 'Magnet' };
          toast({
            title: `${names[powerUp.type]} Activated!`,
            description: "10 seconds",
            className: "bg-black border-purple-500 text-purple-400"
          });
        }
        playSound('powerup');
        createParticles(powerUp.position, '#A855F7', 12, 100);
        return false;
      }
      
      return powerUp.lifetime > 0;
    });
    
    // Update particles
    state.particles = state.particles.filter(particle => {
      particle.position.x += particle.velocity.x * dt;
      particle.position.y += particle.velocity.y * dt;
      particle.lifetime -= dt;
      return particle.lifetime > 0;
    });
    
    // Update power-up timer
    if (state.powerUpActive) {
      state.powerUpActive.duration -= deltaTime;
      if (state.powerUpActive.duration <= 0) {
        state.powerUpActive = null;
      }
    }
    
    // Update combo timer
    if (state.comboTimer > 0) {
      state.comboTimer -= deltaTime;
      if (state.comboTimer <= 0) {
        state.combo = 0;
      }
    }
    
    // Check collisions: bullets vs asteroids
    state.bullets.forEach(bullet => {
      state.asteroids.forEach(asteroid => {
        const dist = getDistance(bullet.position, asteroid.position);
        if (dist < asteroid.radius) {
          destroyAsteroid(asteroid, true);
          state.bullets = state.bullets.filter(b => b.id !== bullet.id);
        }
      });
    });
    
    // Check collisions: ship vs asteroids
    if (!ship.invulnerable) {
      state.asteroids.forEach(asteroid => {
        const dist = getDistance(ship.position, asteroid.position);
        if (dist < SHIP_SIZE / 2 + asteroid.radius) {
          damageShip();
        }
      });
    }
  }, [isPaused, gameOver, targetPosition, isShooting, shoot, destroyAsteroid, damageShip, playSound, createParticles, toast]);

  const render = useCallback(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;
    
    const ctx = canvas.getContext('2d');
    if (!ctx) return;
    
    const state = gameStateRef.current;
    
    // Clear
    ctx.clearRect(0, 0, CANVAS_WIDTH, CANVAS_HEIGHT);
    
    // Draw parallax starfield (3 layers)
    ctx.fillStyle = 'rgba(255, 255, 255, 0.8)';
    for (let layer = 0; layer < 3; layer++) {
      const count = 50 - layer * 10;
      const speed = (layer + 1) * 0.05;
      for (let i = 0; i < count; i++) {
        const x = ((i * 137) % CANVAS_WIDTH + state.survivalTime * speed) % CANVAS_WIDTH;
        const y = ((i * 97) % CANVAS_HEIGHT + state.survivalTime * speed * 0.5) % CANVAS_HEIGHT;
        const size = 1 + layer * 0.5;
        ctx.fillRect(x, y, size, size);
      }
    }
    
    // Draw particles
    state.particles.forEach(particle => {
      const alpha = particle.lifetime;
      ctx.fillStyle = `${particle.color}${Math.floor(alpha * 255).toString(16).padStart(2, '0')}`;
      ctx.beginPath();
      ctx.arc(particle.position.x, particle.position.y, particle.size, 0, Math.PI * 2);
      ctx.fill();
    });
    
    // Draw asteroids
    state.asteroids.forEach(asteroid => {
      ctx.save();
      ctx.translate(asteroid.position.x, asteroid.position.y);
      ctx.rotate(asteroid.rotation);
      
      // 3D effect: scale based on rotation
      const scale = 0.9 + Math.abs(Math.sin(asteroid.rotation)) * 0.2;
      ctx.scale(scale, 1);
      
      // Draw asteroid
      ctx.fillStyle = ASTEROID_COLORS[asteroid.color];
      ctx.strokeStyle = '#000';
      ctx.lineWidth = 2;
      ctx.beginPath();
      
      const sides = 8;
      for (let i = 0; i < sides; i++) {
        const angle = (Math.PI * 2 * i) / sides;
        const variance = 0.8 + Math.random() * 0.4;
        const x = Math.cos(angle) * asteroid.radius * variance;
        const y = Math.sin(angle) * asteroid.radius * variance;
        if (i === 0) ctx.moveTo(x, y);
        else ctx.lineTo(x, y);
      }
      ctx.closePath();
      ctx.fill();
      ctx.stroke();
      
      // Glow for gold asteroids
      if (asteroid.color === 'gold') {
        ctx.shadowBlur = 20;
        ctx.shadowColor = '#FBBF24';
        ctx.stroke();
      }
      
      ctx.restore();
    });
    
    // Draw resources
    state.resources.forEach(resource => {
      const alpha = Math.min(1, resource.lifetime / 2);
      ctx.fillStyle = `${ASTEROID_COLORS[resource.color]}${Math.floor(alpha * 200).toString(16).padStart(2, '0')}`;
      ctx.strokeStyle = '#FFF';
      ctx.lineWidth = 2;
      
      ctx.beginPath();
      ctx.arc(resource.position.x, resource.position.y, 8, 0, Math.PI * 2);
      ctx.fill();
      ctx.stroke();
      
      // Sparkle
      const time = Date.now() / 100;
      ctx.fillStyle = `rgba(255, 255, 255, ${0.5 + Math.sin(time) * 0.5})`;
      ctx.fillRect(resource.position.x - 2, resource.position.y - 2, 4, 4);
    });
    
    // Draw power-ups
    state.powerUps.forEach(powerUp => {
      const alpha = Math.min(1, powerUp.lifetime / 3);
      const colors = {
        shield: '#3B82F6',
        rapidFire: '#EF4444',
        magnet: '#A855F7',
        extraLife: '#10B981',
      };
      
      ctx.fillStyle = `${colors[powerUp.type]}${Math.floor(alpha * 200).toString(16).padStart(2, '0')}`;
      ctx.strokeStyle = '#FFF';
      ctx.lineWidth = 2;
      
      const size = 15 + Math.sin(Date.now() / 200) * 3;
      ctx.beginPath();
      ctx.arc(powerUp.position.x, powerUp.position.y, size, 0, Math.PI * 2);
      ctx.fill();
      ctx.stroke();
      
      // Glow
      ctx.shadowBlur = 15;
      ctx.shadowColor = colors[powerUp.type];
      ctx.stroke();
      ctx.shadowBlur = 0;
    });
    
    // Draw bullets
    state.bullets.forEach(bullet => {
      ctx.fillStyle = '#FFF';
      ctx.shadowBlur = 5;
      ctx.shadowColor = '#FFF';
      ctx.beginPath();
      ctx.arc(bullet.position.x, bullet.position.y, 3, 0, Math.PI * 2);
      ctx.fill();
      ctx.shadowBlur = 0;
      
      // Trail
      ctx.strokeStyle = 'rgba(255, 255, 255, 0.3)';
      ctx.lineWidth = 2;
      ctx.beginPath();
      ctx.moveTo(bullet.position.x, bullet.position.y);
      ctx.lineTo(
        bullet.position.x - bullet.velocity.x * 0.03,
        bullet.position.y - bullet.velocity.y * 0.03
      );
      ctx.stroke();
    });
    
    // Draw ship
    const ship = state.ship;
    ctx.save();
    ctx.translate(ship.position.x, ship.position.y);
    ctx.rotate(ship.rotation);
    
    // Ship body (triangle)
    if (ship.invulnerable && Math.floor(Date.now() / 100) % 2 === 0) {
      ctx.globalAlpha = 0.5;
    }
    
    ctx.fillStyle = '#60A5FA';
    ctx.strokeStyle = '#FFF';
    ctx.lineWidth = 2;
    ctx.beginPath();
    ctx.moveTo(SHIP_SIZE / 2, 0);
    ctx.lineTo(-SHIP_SIZE / 2, SHIP_SIZE / 3);
    ctx.lineTo(-SHIP_SIZE / 3, 0);
    ctx.lineTo(-SHIP_SIZE / 2, -SHIP_SIZE / 3);
    ctx.closePath();
    ctx.fill();
    ctx.stroke();
    
    // Engine glow
    if (targetPosition) {
      ctx.fillStyle = 'rgba(251, 191, 36, 0.8)';
      ctx.beginPath();
      ctx.moveTo(-SHIP_SIZE / 3, SHIP_SIZE / 4);
      ctx.lineTo(-SHIP_SIZE / 2 - 5, 0);
      ctx.lineTo(-SHIP_SIZE / 3, -SHIP_SIZE / 4);
      ctx.fill();
    }
    
    ctx.restore();
  }, []);

  const gameLoop = useCallback((currentTime: number) => {
    if (!lastTimeRef.current) lastTimeRef.current = currentTime;
    const deltaTime = currentTime - lastTimeRef.current;
    lastTimeRef.current = currentTime;
    
    update(deltaTime);
    render();
    setRenderTrigger(prev => prev + 1);
    
    gameLoopRef.current = requestAnimationFrame(gameLoop);
  }, [update, render]);

  // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
  // START/STOP GAME
  // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•

  const startGame = useCallback(() => {
    if (!isConnected) {
      toast({ 
        title: "Wallet Required", 
        description: "Please connect your wallet to play", 
        variant: "destructive" 
      });
      return;
    }
    
    if (!ownsNFT) {
      toast({ 
        title: "Guardian NFT Required", 
        description: "You must own a Guardian NFT to access this game", 
        variant: "destructive" 
      });
      return;
    }
    
    if (playsToday >= gameConfig.maxPlaysPerDay) {
      toast({ 
        title: "Daily Limit Reached", 
        description: `You've used all ${gameConfig.maxPlaysPerDay} plays today. Return tomorrow!`, 
        variant: "destructive" 
      });
      return;
    }

    // Reset game state
    gameStateRef.current = {
      ship: {
        position: { x: CANVAS_WIDTH / 2, y: CANVAS_HEIGHT / 2 },
        velocity: { x: 0, y: 0 },
        rotation: 0,
        shield: 5,
        maxShield: 5,
        lives: 3,
        invulnerable: false,
        invulnerableTime: 0,
      },
      asteroids: [],
      bullets: [],
      resources: [],
      powerUps: [],
      particles: [],
      score: 0,
      combo: 0,
      comboTimer: 0,
      survivalTime: 0,
      asteroidsDestroyed: 0,
      resourcesCollected: 0,
      accuracy: { hits: 0, shots: 0 },
      powerUpActive: null,
    };
    
    // Spawn initial asteroids
    for (let i = 0; i < 3; i++) {
      setTimeout(() => spawnAsteroid(), i * 500);
    }
    
    setGameStarted(true);
    setGameOver(false);
    setShowVictory(false);
    lastTimeRef.current = 0;
    gameLoopRef.current = requestAnimationFrame(gameLoop);
    
    trackEvent('game_start', 'Game', 'Asteroid Mining');
  }, [isConnected, ownsNFT, playsToday, gameConfig.maxPlaysPerDay, spawnAsteroid, gameLoop, toast]);

  const endGame = useCallback(async () => {
    if (!address) return;

    const state = gameStateRef.current;
    const finalScore = Math.min(state.score, gameConfig.scoring.maxScore);
    const survivalMinutes = Math.floor(state.survivalTime / 60000);
    
    // Update stats
    const newStats: GameStats = { ...stats };
    newStats.gamesPlayed++;
    newStats.totalScore += finalScore;
    newStats.totalTime += Math.floor(state.survivalTime / 1000);
    
    if (survivalMinutes >= 5) {
      newStats.gamesWon++;
      newStats.currentStreak++;
      newStats.longestStreak = Math.max(newStats.longestStreak, newStats.currentStreak);
    } else {
      newStats.currentStreak = 0;
    }
    
    if (finalScore > newStats.bestScore) {
      newStats.bestScore = finalScore;
    }
    
    newStats.lastPlayed = Date.now();
    
    GameStorageManager.saveStats('asteroid-mining', address, newStats);
    setStats(newStats);

    try {
      await submitScore(address, 'asteroid-mining', finalScore);
      setPlaysToday(prev => prev + 1);
      trackEvent('game_complete', 'Game', `Asteroid Mining - ${finalScore} pts`);
    } catch (err) {
      console.error('Failed to submit score:', err);
    }
  }, [address, stats, gameConfig.scoring.maxScore, submitScore]);

  // Asteroid spawning interval
  useEffect(() => {
    if (!gameStarted || gameOver || isPaused) return;
    
    const interval = setInterval(() => {
      spawnAsteroid();
    }, ASTEROID_SPAWN_INTERVAL);
    
    return () => clearInterval(interval);
  }, [gameStarted, gameOver, isPaused, spawnAsteroid]);

  // Stop game loop on unmount
  useEffect(() => {
    return () => {
      if (gameLoopRef.current) {
        cancelAnimationFrame(gameLoopRef.current);
      }
    };
  }, []);

  // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
  // RENDER
  // â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•

  if (nftLoading) {
